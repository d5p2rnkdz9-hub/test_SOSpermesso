'use client';
import type { ResultSection } from '@/types/tree';
import { Accordion, AccordionContent, AccordionItem, AccordionTrigger } from '@/components/ui/accordion';

interface FaqAccordionProps {
  sections: ResultSection[];
  substituteVars: (text: string) => string;
}

export function FaqAccordion({ sections, substituteVars }: FaqAccordionProps) {
  // Filter out the lawyer section -- its content is shown in LawyerBanner
  const displaySections = sections.filter(
    (s) => !s.heading.toLowerCase().includes('avvocato'),
  );
  if (displaySections.length === 0) return null;
  return (
    <Accordion type="single" collapsible defaultValue="section-0">
      {displaySections.map((section, index) => (
        <AccordionItem key={index} value={`section-${index}`}>
          <AccordionTrigger>{section.heading}</AccordionTrigger>
          <AccordionContent>
            <p>{substituteVars(section.content)}</p>
          </AccordionContent>
        </AccordionItem>
      ))}
    </Accordion>
  );
}

  heading: string;
  content: string;

  <action>
**Redesign FaqAccordion.tsx** from a Radix accordion to always-open cards with colored left borders.

**1. Remove accordion imports:**
Remove the import of `Accordion, AccordionContent, AccordionItem, AccordionTrigger` from `@/components/ui/accordion`. The component no longer needs Radix accordion.

**2. Define the rotating border colors:**
```typescript
const BORDER_COLORS = ['#42A5F5', '#FFD700', '#FF5252'] as const;
```
Colors rotate: index % 3 maps to blue → gold → red → blue → gold → red...

**3. Replace the JSX with open card layout:**

Keep the existing filtering logic (filter out lawyer/avvocato section).

For each section, render an open card:
```tsx
<div className="flex flex-col gap-4">
  {displaySections.map((section, index) => (
    <div
      key={index}
      className="rounded-[24px] bg-card p-5"
      style={{
        borderInlineStart: `4px solid ${BORDER_COLORS[index % BORDER_COLORS.length]}`,
        boxShadow: '0px 4px 20px rgba(0,0,0,0.08)',
      }}
    >
      <h3 className="text-base font-semibold text-card-foreground">
        {section.heading}
      </h3>
      <p className="mt-2 whitespace-pre-line text-foreground/80">
        {substituteVars(section.content)}
      </p>
    </div>
  ))}
</div>
```

**Key design details:**
- Use `borderInlineStart` (CSS logical property for RTL support) instead of `borderLeft`
- Use `rounded-[24px]` for the sospermesso.it card radius (distinct from app's 12px default)
- Use inline `style` for the colored border since colors are dynamic hex values
- Use `boxShadow` inline style matching sospermesso.it: `0px 4px 20px rgba(0,0,0,0.08)`
- Keep `bg-card` (white) for card background
- Content is always visible -- no collapse/expand interaction
- Heading uses `text-base font-semibold` (same weight as previous AccordionTrigger)
- Content uses `whitespace-pre-line text-foreground/80` (preserves line breaks in content)
- Gap between cards: `gap-4` (16px)

**4. Remove 'use client' if no longer needed:**
Actually, keep `'use client'` since `substituteVars` and the component may be used in client context. (FaqAccordion is rendered inside OutcomePage which is 'use client'.)

**Important:** Use `borderInlineStart` not `borderLeft` for RTL support. This is a CSS logical property that flips to right border in RTL mode.
